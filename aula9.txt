TNo *alocaNo(int k)
{
	TNo *novo=NULL;
	novo = (TNo *)malloc(sizeof(TNo));
	if(novo) //se novo != NULL
	{
		novo->chave = k;
		novo->prox = NULL;
	}

	return novo;
}

void insereInicio(TNo **pprim, int k)
{;

	TNo *novo = NULL;
	novo = alocaNo(k);
	if(novo == NULL) return;

	novo->prox = *pprim;
	*pprim = novo;

}

void imprimir(TNo * paux)
{
	while(paux != NULL)
	{
		printf("\n%d", paux->chave );
		printf("\n%p", paux);
		paux = paux->prox;
	}

}

void removerInicio(TNo **prim)
{
	TNo *paux= *prim;
	if(paux)
	{
		*prim = (*prim)->prox;
		free(paux);
		paux=NULL; // não precisa pq a variavel he local, mas ela quis
	}
}

void removerTudo (TNo **prim)
{
	while(*prim)
	{
		removerInicio(prim);
	}

}
